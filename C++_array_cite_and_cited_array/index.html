C++数组的引用和引用的数组

  • 数组变量array向右结合，相当于(int&)array[n]，在这里array是一个数组，其中数组的每个元素是引用，即引用的数组。
  • 数组的引用中首先和&结合，即(int &array)[n]，这里array是引用，其引用的对象是数组。
  • #include <iostream>

using namespace std;

void Test(int (&array)[5])
{
    cout << "size:" << sizeof(array) << endl;
    for (int i = 0; i < sizeof(array)/4; i++)
    {
      cout << array[i] << " "<<endl;
    }
}
int main()
{
    int  array[5] = {1,2,3,4,5};
    Test(array);
    int n3[3]={2,4,6};
    int (&rn3)[3]=n3;
    for(int i=0;i<3;i++)
      cout<<n3[i]<<"  "<<rn3[i]<<"  "<<endl;
    return 0;
}
