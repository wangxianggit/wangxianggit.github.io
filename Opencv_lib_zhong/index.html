Opencv库添加中文支持

  • 操作环境：windows 10，opencv3.40

  • opencv默认不支持中文输入，需要添加以下代码并调用：

  • putText.h:

    • ~~~
      #ifndef PUTTEXT_H_
      #define PUTTEXT_H_

      #include <windows.h>
      #include
      #include <opencv2/opencv.hpp>

      using namespace cv;

      void GetStringSize(HDC hDC, const char* str, int* w, int* h);
      void putTextZH(Mat &dst, const char* str, Point org, Scalar color, int fontSize,

      const char *fn = "Arial", bool italic = false, bool underline = false);

      #endif // PUTTEXT_H_

  • 调用：

    • #include "putText.h"

void main()
{
    Mat img(150,600,CV_8UC3,Scalar(255,255,255));//初始化图像
    putTextZH(img, "测试", Point(30, 30), Scalar(0, 0, 0), 50, "宋体");
    imwrite("test.png", img);
    imshow("", img);
    waitKey(0);
}
