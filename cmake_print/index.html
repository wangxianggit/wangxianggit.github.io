cmake打印乘法表

  • CMakeLists.txt 文件代码：

    • cmake_minimum_required(VERSION 2.8.11) 
project(test) 
foreach(i RANGE 1 9) 
    set(exp "") 
    foreach(j RANGE 1 ${i}) 
        math(EXPR value "${j} * ${i}") 
        string(APPEND exp "${j}" "*") 
        string(APPEND exp "${i}" "=") 
        string(APPEND exp "${value}" " ") 
    endforeach() 
    message(${exp})
endforeach()
